function binary_search(search_list, guess_numb){
   let first = 0
   let last = search_list.length - 1
   while (first <= last) {
      let mid = (first + last) / 2
      let curr_value = search_list[mid]
      if (curr_value == guess_numb) {
         return mid
      } else if (curr_value > guess_numb) {
         last = mid - 1
      } else {
         first = mid + 1
      }
   }

   return null
}
